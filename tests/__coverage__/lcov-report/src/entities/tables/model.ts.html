export const $tablesProductsTimersOutOfLimits = $tables.map((tables) => {
  const result: Record<string, Record<string, boolean>> = {};
  for (const tableId in tables) {
    result[tableId] = {};
    for (const productId in tables[tableId].products) {
      const product = tables[tableId].products[productId];
      if (product.needTimer) {
        result[tableId][productId] = product.minutesTimer >= product.eachProductUnitMinutesTimer * product.units;
      }
    }
  }
  return result;
});
